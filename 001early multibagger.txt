def detect_sector_clusters(traps_df, sector_map, window_days=5):
    """
    Groups detected traps by sector to find institutional clusters.
    traps_df: Output from your 'detect_traps' function.
    sector_map: A dictionary or CSV mapping Symbol -> Sector.
    """
    # 1. Map symbols to sectors
    traps_df['sector'] = traps_df['symbol'].map(sector_map)
    
    # 2. Group by sector and date window
    # We look for clusters where multiple traps happen in a tight timeframe
    traps_df = traps_df.sort_values('date')
    
    # Count traps per sector in a rolling window
    sector_clusters = traps_df.groupby('sector').rolling(
        window=f'{window_days}D', on='date'
    )['symbol'].count().reset_index()
    
    sector_clusters.rename(columns={'symbol': 'trap_count'}, inplace=True)
    
    # 3. Filter for "Hot Sectors" (e.g., 3+ traps in 5 days)
    hot_sectors = sector_clusters[sector_clusters['trap_count'] >= 3]
    
    return hot_sectors
